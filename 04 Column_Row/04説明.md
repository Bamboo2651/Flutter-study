void main() {
  final col = Column(
    mainAxisAlignment: MainAxisAlignment.center,
    crossAxisAlignment: CrossAxisAlignment.center,
    children: [Text("レモン"), Text("リンゴ"), Text("ブドウ")],
  );
  final a = MaterialApp(
    home: Scaffold(body: Center(child: col)),
  );
